            close(fd);
            execvp("/usr/bin/xxd", argva);
            perror("Error executing xxd");
          }
          else
          {
            perror("Error redirecting output to file");
            close(fd);
          }
          return 1;
        }
        else
        {
          perror("Error opening output file");
          return 1;
        }
      }
      else
      {
        fwrite("Not every wizards can use or destroy a Horcrux!\n", 1uLL, 0x30uLL, stderr);
        return 1;
      }
    }
